nota1 = Number(prompt("Digite a primeira nota: ")) nota2 = Number(prompt("Digite a segunda nota: ")) nota3 = Number(prompt("Digite a terceira nota: ")) nota4 = Number(prompt("Digite a quarta nota: ")) nome = prompt("Digite o nome do aluno: ") media = (nota1+nota2+nota3+nota4)/4 if (media >= 5) { alert(`Aluno Aprovado ${nome} - media ${media}`) } else { alert(`Aluno Reprovado ${nome} - media ${media}`) }
